    <section id="allure-why" class="paper-section paper-anchor">
      <h2>Why it matters</h2>
      <ul class="paper-list">
        <li>AI systems like DeepCubeA can solve the Rubik's Cube but do not explain their strategies in human-understandable terms.</li>
        <li>Educational settings need step-by-step reasoning, not only final solved states.</li>
        <li>Children benefit from interactive, visual, and guided learning interfaces for complex problem-solving tasks.</li>
        <li>There is a gap between automated planning/search systems and explainable, collaborative AI built for learning.</li>
      </ul>

      <figure class="paper-figure paper-figure-narrow">

        <figcaption>Figure 2. The white cross subgoal used as a demonstration case.</figcaption>
      </figure>

      <p class="paper-reference">The system first focuses on forming the white cross (Figure 2), a standard early subgoal.</p>
    </section>

    <section id="allure-what" class="paper-section paper-anchor">
      <h2>What we did</h2>
      <ul class="paper-list">
        <li>Introduced <strong>ALLURE</strong>, a multi-modal collaborative AI system for learning to solve the Rubik's Cube.</li>
        <li>Extended DeepCubeA with hindsight experience replay so it can achieve arbitrary subgoals (for example, forming the white cross).</li>
        <li>Used inductive logic programming (<strong>Popper ILP</strong>) to extract human-understandable algorithms from solution traces.</li>
        <li>Translated logic-based rules into natural-language instructional explanations.</li>
        <li>Built a 3D interactive cube visualization and a Rasa-based chatbot for guided, explainable learning.</li>
        <li>Enabled collaborative use where learners can propose their own subgoals or example algorithms for refinement.</li>
      </ul>
    </section>

    <section id="allure-how" class="paper-section paper-anchor">
      <h2>How it works</h2>
      <ul class="paper-list">
        <li><strong>Goal-conditioned learning:</strong> a DeepCubeA variant trained with hindsight experience replay achieves arbitrary cube subgoals.</li>
        <li><strong>Example extraction:</strong> solution traces between subgoals are collected as state-action examples.</li>
        <li><strong>Inductive logic programming:</strong> Popper ILP induces concise first-order logic programs that describe the move algorithms.</li>
        <li><strong>Natural-language translation:</strong> induced logic rules are rendered as explanation templates for learners.</li>
        <li><strong>Multi-modal interface:</strong> 3D cube visualization plus chatbot interaction (text, animation, and voice) supports guided learning.</li>
      </ul>

  <script>
    (() => {
      const attachCitationAnchor = () => {
        const citationParagraph = Array.from(document.querySelectorAll(".page__content p")).find(
          (paragraph) => paragraph.textContent.includes("Recommended citation")
        );
        if (citationParagraph && !citationParagraph.id) {
          citationParagraph.id = "paper-citation";
        }
      };

      attachCitationAnchor();
      window.requestAnimationFrame(attachCitationAnchor);
      if (document.readyState === "loading") {
        document.addEventListener("DOMContentLoaded", attachCitationAnchor, { once: true });
      }
      window.addEventListener("load", attachCitationAnchor, { once: true });

      const canvas = document.getElementById("allure-cube-bg");
      if (!canvas) {
        return;
      }

      const ctx = canvas.getContext("2d", { alpha: true });
      if (!ctx) {
        return;
      }

      const root = document.documentElement;
      const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)");
      const moveLabels = ["D'", "F'", "R", "F"];
      const stateXRatios = [0.12, 0.32, 0.52, 0.72, 0.88];
      const stateYRatios = [0.56, 0.47, 0.58, 0.49, 0.44];
      const cubeColors = [
        [222, 63, 61],
        [59, 130, 246],
        [53, 176, 112],
        [242, 154, 46],
        [243, 214, 80],
        [238, 238, 238]
      ];

      let width = 0;
      let height = 0;
      let dpr = 1;
      let palette = null;
      let states = [];
      let paths = [];
      let packets = [];
      let floatingCubes = [];
      let animationId = null;
      let lastTime = 0;
      let packetTimer = 0;
      let packetIndex = 0;

      const parseColor = (value) => {
        if (!value) {
          return null;
        }
        const trimmed = value.trim();
        if (trimmed.startsWith("#")) {
          const hex = trimmed.slice(1);
          if (hex.length === 3) {
            return hex.split("").map((char) => parseInt(char + char, 16));
          }
          if (hex.length === 6) {
            return [
              parseInt(hex.slice(0, 2), 16),
              parseInt(hex.slice(2, 4), 16),
              parseInt(hex.slice(4, 6), 16)
            ];
          }
        }
        const rgbMatch = trimmed.match(/rgba?\(([^)]+)\)/);
        if (rgbMatch) {
          const parts = rgbMatch[1].split(",").map((part) => parseFloat(part.trim()));
          if (parts.length >= 3) {
            return parts.slice(0, 3);
          }
        }
        return null;
      };

      const mixColor = (base, accent, amount) =>
        base.map((channel, index) => Math.round(channel + (accent[index] - channel) * amount));

      const toRgba = (rgb, alpha) => `rgba(${rgb[0]}, ${rgb[1]}, ${rgb[2]}, ${alpha})`;
      const rand = (min, max) => min + Math.random() * (max - min);
      const clamp = (value, min, max) => Math.max(min, Math.min(max, value));

      const roundedRect = (x, y, w, h, r) => {
        const radius = Math.min(r, w / 2, h / 2);
        ctx.beginPath();
        ctx.moveTo(x + radius, y);
        ctx.lineTo(x + w - radius, y);
        ctx.quadraticCurveTo(x + w, y, x + w, y + radius);
        ctx.lineTo(x + w, y + h - radius);
        ctx.quadraticCurveTo(x + w, y + h, x + w - radius, y + h);
        ctx.lineTo(x + radius, y + h);
        ctx.quadraticCurveTo(x, y + h, x, y + h - radius);
        ctx.lineTo(x, y + radius);
        ctx.quadraticCurveTo(x, y, x + radius, y);
        ctx.closePath();
      };

      const quadAt = (path, t) => {
        const inv = 1 - t;
        return {
          x: inv * inv * path.p0.x + 2 * inv * t * path.cp.x + t * t * path.p1.x,
          y: inv * inv * path.p0.y + 2 * inv * t * path.cp.y + t * t * path.p1.y
        };
      };

      const updatePalette = () => {
        const styles = getComputedStyle(root);
        const bgColor = parseColor(styles.getPropertyValue("--global-bg-color")) || [243, 244, 246];
        const accentColor = parseColor(styles.getPropertyValue("--global-link-color")) || [82, 173, 200];
        const textColor = parseColor(styles.getPropertyValue("--global-text-color")) || [45, 51, 58];

        palette = {
          wash: mixColor(bgColor, accentColor, 0.2),
          stateStroke: mixColor(bgColor, accentColor, 0.5),
          stateGlow: mixColor(bgColor, accentColor, 0.32),
          pathSoft: mixColor(bgColor, accentColor, 0.35),
          pathStrong: mixColor(bgColor, accentColor, 0.58),
          packetText: mixColor(textColor, [255, 255, 255], 0.18),
          cubeFrame: mixColor(bgColor, textColor, 0.25),
          cubePanel: mixColor(bgColor, textColor, 0.12),
          cubeFaces: cubeColors.map((color) => mixColor(bgColor, color, 0.72)),
          cubeFaceSoft: cubeColors.map((color) => mixColor(bgColor, color, 0.46))
        };
      };

      const buildSceneGeometry = () => {
        states = stateXRatios.map((xRatio, index) => ({
          x: width * xRatio,
          y: height * stateYRatios[index],
          size: clamp(Math.min(width, height) * 0.08, 38, 68),
          goal: index === stateXRatios.length - 1,
          stickers: buildStickers(index === stateXRatios.length - 1)
        }));

        paths = states.slice(0, -1).map((state, index) => {
          const next = states[index + 1];
          const controlYOffset = index % 2 === 0 ? -height * 0.12 : height * 0.12;
          return {
            p0: { x: state.x, y: state.y },
            p1: { x: next.x, y: next.y },
            cp: { x: (state.x + next.x) * 0.5, y: (state.y + next.y) * 0.5 + controlYOffset }
          };
        });
      };

      const buildStickers = (goal = false) => {
        const stickers = Array.from({ length: 9 }, () => Math.floor(Math.random() * 5));
        if (goal) {
          const whiteIndex = 5;
          stickers[1] = whiteIndex;
          stickers[3] = whiteIndex;
          stickers[4] = whiteIndex;
          stickers[5] = whiteIndex;
          stickers[7] = whiteIndex;
        }
        return stickers;
      };

      const initFloatingCubes = () => {
        const count = Math.max(10, Math.min(24, Math.floor((width * height) / 65000)));
        floatingCubes = Array.from({ length: count }, () => ({
          x: rand(0, width),
          y: rand(0, height),
          size: rand(14, 28),
          vx: rand(-14, 14),
          vy: rand(10, 28),
          rot: rand(0, Math.PI * 2),
          vr: rand(-0.34, 0.34),
          alpha: rand(0.09, 0.2),
          stickers: buildStickers(false)
        }));
      };

      const resize = () => {
        width = window.innerWidth;
        height = window.innerHeight;
        dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
        canvas.width = width * dpr;
        canvas.height = height * dpr;
        canvas.style.width = `${width}px`;
        canvas.style.height = `${height}px`;
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
        updatePalette();
        buildSceneGeometry();
        initFloatingCubes();
      };

      const spawnPacket = () => {
        packets.push({
          pathIndex: packetIndex % paths.length,
          t: 0,
          speed: rand(0.00023, 0.00034),
          size: rand(13, 18),
          label: moveLabels[packetIndex % moveLabels.length],
          colorIndex: packetIndex % palette.cubeFaces.length
        });
        packetIndex += 1;
      };

      const drawStateCube = (state, phase) => {
        const panel = state.size;
        const gap = panel * 0.06;
        const inner = panel * 0.74;
        const cell = (inner - gap * 2) / 3;
        const start = -inner / 2;
        const stickers = state.stickers;

        ctx.save();
        ctx.translate(state.x, state.y);
        ctx.rotate(Math.sin(phase + state.x * 0.003) * 0.08);
        roundedRect(-panel / 2, -panel / 2, panel, panel, panel * 0.13);
        ctx.fillStyle = toRgba(palette.cubePanel, 0.48);
        ctx.fill();
        ctx.strokeStyle = toRgba(palette.stateStroke, 0.5);
        ctx.lineWidth = 1;
        ctx.stroke();

        for (let row = 0; row < 3; row += 1) {
          for (let col = 0; col < 3; col += 1) {
            const sticker = stickers[row * 3 + col];
            const color = state.goal ? palette.cubeFaces[sticker] : palette.cubeFaceSoft[sticker];
            const x = start + col * (cell + gap);
            const y = start + row * (cell + gap);
            roundedRect(x, y, cell, cell, cell * 0.18);
            ctx.fillStyle = toRgba(color, state.goal ? 0.8 : 0.58);
            ctx.fill();
          }
        }
        ctx.restore();
      };

      const drawFloatingCube = (cube) => {
        const panel = cube.size;
        const gap = panel * 0.08;
        const inner = panel * 0.72;
        const cell = (inner - gap * 2) / 3;
        const start = -inner / 2;

        ctx.save();
        ctx.translate(cube.x, cube.y);
        ctx.rotate(cube.rot);
        roundedRect(-panel / 2, -panel / 2, panel, panel, panel * 0.14);
        ctx.fillStyle = toRgba(palette.cubePanel, cube.alpha * 0.8);
        ctx.fill();
        ctx.strokeStyle = toRgba(palette.cubeFrame, cube.alpha);
        ctx.lineWidth = 0.8;
        ctx.stroke();

        for (let row = 0; row < 3; row += 1) {
          for (let col = 0; col < 3; col += 1) {
            const color = palette.cubeFaceSoft[cube.stickers[row * 3 + col]];
            roundedRect(start + col * (cell + gap), start + row * (cell + gap), cell, cell, cell * 0.2);
            ctx.fillStyle = toRgba(color, cube.alpha * 1.12);
            ctx.fill();
          }
        }
        ctx.restore();
      };

      const drawPath = (path) => {
        ctx.beginPath();
        ctx.moveTo(path.p0.x, path.p0.y);
        ctx.quadraticCurveTo(path.cp.x, path.cp.y, path.p1.x, path.p1.y);
      };

      const drawArrow = (path) => {
        const tip = quadAt(path, 1);
        const tail = quadAt(path, 0.96);
        const angle = Math.atan2(tip.y - tail.y, tip.x - tail.x);
        const size = 8;

        ctx.beginPath();
        ctx.moveTo(tip.x, tip.y);
        ctx.lineTo(tip.x - Math.cos(angle - Math.PI / 8) * size, tip.y - Math.sin(angle - Math.PI / 8) * size);
        ctx.lineTo(tip.x - Math.cos(angle + Math.PI / 8) * size, tip.y - Math.sin(angle + Math.PI / 8) * size);
        ctx.closePath();
        ctx.fillStyle = toRgba(palette.pathStrong, 0.35);
        ctx.fill();
      };

      const drawScene = (phase) => {
        ctx.clearRect(0, 0, width, height);

        const wash = ctx.createRadialGradient(
          width * 0.52,
          height * 0.5,
          30,
          width * 0.5,
          height * 0.5,
          Math.max(width, height) * 0.75
        );
        wash.addColorStop(0, toRgba(palette.wash, 0.09));
        wash.addColorStop(1, "rgba(0, 0, 0, 0)");
        ctx.fillStyle = wash;
        ctx.fillRect(0, 0, width, height);

        floatingCubes.forEach((cube) => drawFloatingCube(cube));

        ctx.lineCap = "round";
        paths.forEach((path) => {
          drawPath(path);
          ctx.strokeStyle = toRgba(palette.pathSoft, 0.16);
          ctx.lineWidth = 12;
          ctx.stroke();

          drawPath(path);
          ctx.strokeStyle = toRgba(palette.pathStrong, 0.28);
          ctx.lineWidth = 2.1;
          ctx.stroke();
          drawArrow(path);
        });

        states.forEach((state) => {
          const glowRadius = state.size * 0.95;
          const glow = ctx.createRadialGradient(state.x, state.y, state.size * 0.18, state.x, state.y, glowRadius);
          glow.addColorStop(0, toRgba(palette.stateGlow, state.goal ? 0.34 : 0.18));
          glow.addColorStop(1, "rgba(0, 0, 0, 0)");
          ctx.fillStyle = glow;
          ctx.beginPath();
          ctx.arc(state.x, state.y, glowRadius, 0, Math.PI * 2);
          ctx.fill();
          drawStateCube(state, phase);
        });

        packets.forEach((packet) => {
          const path = paths[packet.pathIndex];
          const point = quadAt(path, packet.t);
          const face = palette.cubeFaces[packet.colorIndex];
          const glow = ctx.createRadialGradient(point.x, point.y, 2, point.x, point.y, packet.size * 1.8);
          glow.addColorStop(0, toRgba(face, 0.55));
          glow.addColorStop(1, "rgba(0, 0, 0, 0)");
          ctx.fillStyle = glow;
          ctx.beginPath();
          ctx.arc(point.x, point.y, packet.size * 1.8, 0, Math.PI * 2);
          ctx.fill();

          ctx.beginPath();
          ctx.arc(point.x, point.y, packet.size, 0, Math.PI * 2);
          ctx.fillStyle = toRgba(face, 0.78);
          ctx.fill();
          ctx.lineWidth = 1;
          ctx.strokeStyle = toRgba(palette.cubeFrame, 0.55);
          ctx.stroke();

          ctx.fillStyle = toRgba(palette.packetText, 0.9);
          ctx.font = `${Math.max(11, packet.size * 0.68)}px system-ui, sans-serif`;
          ctx.textAlign = "center";
          ctx.textBaseline = "middle";
          ctx.fillText(packet.label, point.x, point.y + 0.5);
        });
      };

      const step = (timestamp) => {
        if (!lastTime) {
          lastTime = timestamp;
        }
        const dt = timestamp - lastTime;
        lastTime = timestamp;

        packetTimer += dt;
        if (packetTimer > 720) {
          packetTimer = 0;
          spawnPacket();
        }

        packets = packets
          .map((packet) => ({ ...packet, t: packet.t + packet.speed * dt }))
          .filter((packet) => packet.t <= 1.05);

        floatingCubes.forEach((cube) => {
          cube.x += (cube.vx * dt) / 1000;
          cube.y += (cube.vy * dt) / 1000;
          cube.rot += (cube.vr * dt) / 1000;

          if (cube.y > height + cube.size) {
            cube.y = -cube.size;
            cube.x = rand(0, width);
          }
          if (cube.x > width + cube.size) {
            cube.x = -cube.size;
          }
          if (cube.x < -cube.size) {
            cube.x = width + cube.size;
          }
        });

        drawScene(timestamp * 0.001);
        animationId = window.requestAnimationFrame(step);
      };

      const stop = () => {
        if (animationId !== null) {
          window.cancelAnimationFrame(animationId);
          animationId = null;
        }
      };

      const start = () => {
        stop();
        lastTime = 0;
        drawScene(0);
        if (!prefersReducedMotion.matches) {
          animationId = window.requestAnimationFrame(step);
        }
      };

      const handleMotionChange = () => {
        start();
      };

      const observeTheme = () => {
        const observer = new MutationObserver(() => {
          updatePalette();
          drawScene(0);
        });
        observer.observe(root, { attributes: true, attributeFilter: ["data-theme"] });
      };

      resize();
      for (let i = 0; i < 4; i += 1) {
        spawnPacket();
      }
      drawScene(0);
      start();
      observeTheme();

      window.addEventListener("resize", () => {
        resize();
        drawScene(0);
      });

      if (typeof prefersReducedMotion.addEventListener === "function") {
        prefersReducedMotion.addEventListener("change", handleMotionChange);
      } else if (typeof prefersReducedMotion.addListener === "function") {
        prefersReducedMotion.addListener(handleMotionChange);
      }
    })();
  </script>
</div>
