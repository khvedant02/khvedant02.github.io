    <section id="gear-why" class="paper-section paper-anchor">
      <h2>Why it matters</h2>
      <ul class="paper-list">
        <li>Systematic reviews are time-intensive and require precise query formulation.</li>
        <li>Researchers depend on expert librarians to define research questions and develop structured search protocols.</li>
        <li>Query development and translation can affect reproducibility and search quality.</li>
        <li>Librarian time and bandwidth are limited, creating a need for AI-assisted support in early systematic review stages.</li>
      </ul>
    </section>

    <section id="gear-what" class="paper-section paper-anchor">
      <h2>What we did</h2>
      <ul class="paper-list">
        <li>We introduce <strong>GEAR-Up</strong>, a modular pipeline that assists librarians in systematic review query construction.</li>
        <li>We expand natural language queries using knowledge graphs (for example, PubMed) and pretrained masked language models.</li>
        <li>We generate related reformulated queries using ChatGPT prompted with expanded concepts and relations.</li>
        <li>We retrieve and rank relevant articles using PubMed searches combined with a FAISS-powered retriever.</li>
        <li>We evaluate retrieved results qualitatively with an in-house librarian, reporting favorable outcomes in reducing librarian burden and improving search quality.</li>
      </ul>
    </section>

    <section id="gear-how" class="paper-section paper-anchor">
      <h2>How it works</h2>
      <ul class="paper-list">
        <li><strong>Query Expansion Module:</strong> extracts seed concepts from a natural language query and augments them using knowledge graphs (for example, PubMed) and masked language models.</li>
        <li><strong>Related Query Generation Module:</strong> prompts ChatGPT with expanded concepts and relationships to generate reformulated queries.</li>
        <li><strong>Article Search and Retrieval Module:</strong> executes PubMed searches and applies a FAISS-powered retriever to narrow results to the most relevant titles, abstracts, and passages.</li>
        <li><strong>Librarian-in-the-loop refinement:</strong> incorporates librarian feedback for relevance, safety, ethics, and bias controls.</li>
      </ul>

    <section id="gear-key" class="paper-section paper-anchor">
      <h2>Key contributions</h2>
      <ul class="paper-list">
        <li>A modular architecture integrating generative AI with external knowledge graphs for systematic review query construction.</li>
        <li>Automation of the first two systematic review stages: issue identification and protocol-based query formulation.</li>
        <li>Integration of FAISS-based dense retrieval for narrowing PubMed search results.</li>
        <li>Qualitative evaluation with an in-house librarian showing reduced burden and improved search quality.</li>
      </ul>

  <script>
    (() => {
      const attachCitationAnchor = () => {
        const citationParagraph = Array.from(document.querySelectorAll(".page__content p")).find(
          (paragraph) => paragraph.textContent.includes("Recommended citation")
        );
        if (citationParagraph && !citationParagraph.id) {
          citationParagraph.id = "paper-citation";
        }
      };

      attachCitationAnchor();
      window.requestAnimationFrame(attachCitationAnchor);
      if (document.readyState === "loading") {
        document.addEventListener("DOMContentLoaded", attachCitationAnchor, { once: true });
      }
      window.addEventListener("load", attachCitationAnchor, { once: true });

      const canvas = document.getElementById("gear-pipeline-bg");
      if (!canvas) {
        return;
      }

      const ctx = canvas.getContext("2d", { alpha: true });
      if (!ctx) {
        return;
      }

      const root = document.documentElement;
      const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)");

      let width = 0;
      let height = 0;
      let dpr = 1;
      let palette = null;
      let animationId = null;
      let lastFrame = 0;
      let spawnTimer = 0;

      let packets = [];
      let effects = [];
      let rings = [];

      const stageRatios = [0.1, 0.34, 0.58, 0.85];
      const stageLabels = ["Seed Query", "Expand", "Generate", "Retrieve"];
      const laneRatios = [0.36, 0.5, 0.64];

      const clamp = (value, min, max) => Math.max(min, Math.min(max, value));
      const rand = (min, max) => min + Math.random() * (max - min);
      const lerp = (start, end, t) => start + (end - start) * t;

      const parseColor = (value) => {
        if (!value) {
          return null;
        }
        const trimmed = value.trim();
        if (trimmed.startsWith("#")) {
          const hex = trimmed.slice(1);
          if (hex.length === 3) {
            return hex.split("").map((char) => parseInt(char + char, 16));
          }
          if (hex.length === 6) {
            return [
              parseInt(hex.slice(0, 2), 16),
              parseInt(hex.slice(2, 4), 16),
              parseInt(hex.slice(4, 6), 16)
            ];
          }
        }
        const rgbMatch = trimmed.match(/rgba?\(([^)]+)\)/);
        if (rgbMatch) {
          const parts = rgbMatch[1].split(",").map((part) => parseFloat(part.trim()));
          if (parts.length >= 3) {
            return parts.slice(0, 3);
          }
        }
        return null;
      };

      const toRgba = (rgb, alpha) => `rgba(${rgb[0]}, ${rgb[1]}, ${rgb[2]}, ${alpha})`;

      const mixColor = (base, accent, amount) =>
        base.map((channel, index) => Math.round(channel + (accent[index] - channel) * amount));

      const updatePalette = () => {
        const styles = getComputedStyle(root);
        const bg = parseColor(styles.getPropertyValue("--global-bg-color")) || [242, 242, 242];
        const accent = parseColor(styles.getPropertyValue("--global-link-color")) || [75, 146, 177];
        const text = parseColor(styles.getPropertyValue("--global-text-color")) || [54, 58, 66];

        palette = {
          wash: mixColor(bg, accent, 0.24),
          lane: mixColor(bg, accent, 0.34),
          laneBorder: mixColor(bg, text, 0.3),
          stageFill: mixColor(bg, text, 0.22),
          stageBorder: mixColor(bg, text, 0.42),
          stageText: mixColor(bg, text, 0.72),
          packet: mixColor(bg, accent, 0.78),
          packetDim: mixColor(bg, text, 0.58),
          rank: mixColor(bg, accent, 0.94),
          relation: mixColor(bg, accent, 0.66),
          burst: mixColor(bg, accent, 0.86)
        };
      };

      const stageX = (index) => width * stageRatios[index];
      const laneY = (index) => height * laneRatios[index];

      const topAt = (xRatio) => {
        const bend = Math.sin((xRatio - 0.1) * Math.PI * 1.15) * 0.03;
        return height * (0.26 + bend);
      };

      const bottomAt = (xRatio) => {
        const bend = Math.sin((xRatio - 0.1) * Math.PI * 1.15) * 0.03;
        return height * (0.74 - bend);
      };

      const emitExpansionBurst = (x, y) => {
        for (let i = 0; i < 7; i += 1) {
          effects.push({
            kind: "orbit",
            cx: x,
            cy: y,
            radius: rand(18, 46),
            angle: rand(0, Math.PI * 2),
            speed: rand(0.65, 1.45) * (Math.random() > 0.5 ? 1 : -1),
            life: 0,
            duration: rand(1000, 1700),
            alpha: rand(0.14, 0.3)
          });
        }
      };

      const emitGenerationTrail = (x, y) => {
        for (let i = 0; i < 3; i += 1) {
          effects.push({
            kind: "streak",
            x,
            y: y + rand(-14, 14),
            vx: rand(64, 110),
            vy: rand(-18, 18),
            life: 0,
            duration: rand(540, 900),
            alpha: rand(0.12, 0.24)
          });
        }
      };

      const emitRankRing = (x, y) => {
        rings.push({
          x,
          y,
          r: 3,
          vr: rand(44, 72),
          life: 0,
          duration: rand(620, 980),
          alpha: rand(0.2, 0.35)
        });
      };

      const spawnPacket = () => {
        if (packets.length > 44) {
          return;
        }
        const lane = Math.floor(rand(0, laneRatios.length));
        packets.push({
          x: stageX(0) - rand(18, 34),
          y: laneY(lane) + rand(-8, 8),
          lane,
          vx: rand(66, 108),
          r: rand(1.4, 2.4),
          alpha: rand(0.22, 0.42),
          wobble: rand(0, Math.PI * 2),
          expanded: false,
          generated: false,
          ranked: false
        });
      };

      const splitPacket = (packet) => {
        const candidates = [0, 1, 2].filter((index) => index !== packet.lane);
        const targetLane = candidates[Math.floor(rand(0, candidates.length))];
        packets.push({
          x: packet.x - rand(3, 10),
          y: packet.y + rand(-8, 8),
          lane: targetLane,
          vx: packet.vx * rand(0.9, 1.08),
          r: clamp(packet.r * rand(0.85, 1.15), 1.2, 2.7),
          alpha: clamp(packet.alpha * rand(0.84, 1.16), 0.16, 0.48),
          wobble: rand(0, Math.PI * 2),
          expanded: true,
          generated: true,
          ranked: false
        });
      };

      const updatePackets = (dt, nowMs) => {
        const expansionX = stageX(1);
        const generationX = stageX(2);
        const retrievalX = stageX(3);

        for (let i = packets.length - 1; i >= 0; i -= 1) {
          const packet = packets[i];
          packet.x += packet.vx * dt;

          const targetY = laneY(packet.lane);
          packet.y += (targetY - packet.y) * 0.12;
          packet.y += Math.sin(nowMs * 0.0025 + packet.wobble) * 0.18;

          if (!packet.expanded && packet.x >= expansionX) {
            packet.expanded = true;
            emitExpansionBurst(expansionX, packet.y);
          }

          if (!packet.generated && packet.x >= generationX) {
            packet.generated = true;
            emitGenerationTrail(generationX, packet.y);
            if (Math.random() > 0.46) {
              splitPacket(packet);
            }
          }

          if (!packet.ranked && packet.x >= retrievalX) {
            packet.ranked = true;
            emitRankRing(packet.x, packet.y);
          }

          if (packet.x > width + 30) {
            packets.splice(i, 1);
          }
        }
      };

      const updateEffects = (dt) => {
        for (let i = effects.length - 1; i >= 0; i -= 1) {
          const effect = effects[i];
          effect.life += dt * 1000;
          if (effect.kind === "streak") {
            effect.x += effect.vx * dt;
            effect.y += effect.vy * dt;
          }
          if (effect.life >= effect.duration) {
            effects.splice(i, 1);
          }
        }

        for (let i = rings.length - 1; i >= 0; i -= 1) {
          const ring = rings[i];
          ring.life += dt * 1000;
          ring.r += ring.vr * dt;
          if (ring.life >= ring.duration) {
            rings.splice(i, 1);
          }
        }
      };

      const drawLane = () => {
        const steps = 26;
        const start = stageRatios[0] - 0.045;
        const end = stageRatios[3] + 0.08;

        ctx.beginPath();
        for (let i = 0; i <= steps; i += 1) {
          const t = i / steps;
          const xRatio = lerp(start, end, t);
          const x = width * xRatio;
          const y = topAt(xRatio);
          if (i === 0) {
            ctx.moveTo(x, y);
          } else {
            ctx.lineTo(x, y);
          }
        }
        for (let i = steps; i >= 0; i -= 1) {
          const t = i / steps;
          const xRatio = lerp(start, end, t);
          const x = width * xRatio;
          const y = bottomAt(xRatio);
          ctx.lineTo(x, y);
        }
        ctx.closePath();
        ctx.fillStyle = toRgba(palette.lane, 0.1);
        ctx.fill();

        ctx.strokeStyle = toRgba(palette.laneBorder, 0.22);
        ctx.lineWidth = 1.4;
        ctx.beginPath();
        for (let i = 0; i <= steps; i += 1) {
          const t = i / steps;
          const xRatio = lerp(start, end, t);
          const x = width * xRatio;
          const y = topAt(xRatio);
          if (i === 0) {
            ctx.moveTo(x, y);
          } else {
            ctx.lineTo(x, y);
          }
        }
        ctx.stroke();

        ctx.beginPath();
        for (let i = 0; i <= steps; i += 1) {
          const t = i / steps;
          const xRatio = lerp(start, end, t);
          const x = width * xRatio;
          const y = bottomAt(xRatio);
          if (i === 0) {
            ctx.moveTo(x, y);
          } else {
            ctx.lineTo(x, y);
          }
        }
        ctx.stroke();
      };

      const drawRelations = (nowMs) => {
        const x1 = stageX(1);
        const x2 = stageX(2);
        const yCenter = laneY(1);

        for (let i = 0; i < 6; i += 1) {
          const offset = (i - 2.5) * 16;
          const wobble = Math.sin(nowMs * 0.0018 + i * 0.8) * 7;
          ctx.strokeStyle = toRgba(palette.relation, 0.1 + i * 0.012);
          ctx.lineWidth = 1;
          ctx.beginPath();
          ctx.moveTo(x1, yCenter + offset);
          ctx.quadraticCurveTo((x1 + x2) * 0.5, yCenter + wobble, x2, yCenter + offset * 0.65);
          ctx.stroke();
        }
      };

      const drawRoundedRectPath = (x, y, w, h, r) => {
        const radius = Math.min(r, w * 0.5, h * 0.5);
        if (typeof ctx.roundRect === "function") {
          ctx.beginPath();
          ctx.roundRect(x, y, w, h, radius);
          return;
        }
        ctx.beginPath();
        ctx.moveTo(x + radius, y);
        ctx.lineTo(x + w - radius, y);
        ctx.quadraticCurveTo(x + w, y, x + w, y + radius);
        ctx.lineTo(x + w, y + h - radius);
        ctx.quadraticCurveTo(x + w, y + h, x + w - radius, y + h);
        ctx.lineTo(x + radius, y + h);
        ctx.quadraticCurveTo(x, y + h, x, y + h - radius);
        ctx.lineTo(x, y + radius);
        ctx.quadraticCurveTo(x, y, x + radius, y);
        ctx.closePath();
      };

      const drawStages = () => {
        ctx.font = "600 11px ui-sans-serif, system-ui, sans-serif";
        ctx.textAlign = "center";
        ctx.textBaseline = "middle";

        stageRatios.forEach((ratio, index) => {
          const x = width * ratio;
          const y = laneY(1);
          const boxW = index === 3 ? 84 : 88;
          const boxH = 26;

          ctx.fillStyle = toRgba(palette.stageFill, 0.13);
          ctx.strokeStyle = toRgba(palette.stageBorder, 0.26);
          ctx.lineWidth = 1;
          drawRoundedRectPath(x - boxW / 2, y - boxH / 2, boxW, boxH, 8);
          ctx.fill();
          ctx.stroke();

          ctx.fillStyle = toRgba(palette.stageText, 0.58);
          ctx.fillText(stageLabels[index], x, y + 0.5);

          ctx.beginPath();
          ctx.arc(x, y, 2.1, 0, Math.PI * 2);
          ctx.fillStyle = toRgba(palette.stageBorder, 0.45);
          ctx.fill();
        });
      };

      const drawPackets = () => {
        packets.forEach((packet) => {
          const color = packet.ranked ? palette.rank : packet.generated ? palette.packet : palette.packetDim;
          const alpha = packet.ranked ? 0.42 : packet.alpha;
          ctx.beginPath();
          ctx.arc(packet.x, packet.y, packet.r, 0, Math.PI * 2);
          ctx.fillStyle = toRgba(color, alpha);
          ctx.fill();
        });
      };

      const drawEffects = () => {
        effects.forEach((effect) => {
          const t = clamp(effect.life / effect.duration, 0, 1);
          const fade = 1 - t;

          if (effect.kind === "orbit") {
            const angle = effect.angle + effect.speed * t * Math.PI * 2;
            const x = effect.cx + Math.cos(angle) * effect.radius;
            const y = effect.cy + Math.sin(angle) * effect.radius * 0.56;
            ctx.beginPath();
            ctx.arc(x, y, 1.35, 0, Math.PI * 2);
            ctx.fillStyle = toRgba(palette.burst, effect.alpha * fade);
            ctx.fill();
            return;
          }

          if (effect.kind === "streak") {
            const x2 = effect.x - 8;
            const y2 = effect.y - effect.vy * 0.03;
            ctx.strokeStyle = toRgba(palette.packet, effect.alpha * fade);
            ctx.lineWidth = 1.2;
            ctx.beginPath();
            ctx.moveTo(effect.x, effect.y);
            ctx.lineTo(x2, y2);
            ctx.stroke();
          }
        });

        rings.forEach((ring) => {
          const t = clamp(ring.life / ring.duration, 0, 1);
          ctx.strokeStyle = toRgba(palette.rank, ring.alpha * (1 - t));
          ctx.lineWidth = 1.4;
          ctx.beginPath();
          ctx.arc(ring.x, ring.y, ring.r, 0, Math.PI * 2);
          ctx.stroke();
        });
      };

      const drawBackgroundWash = () => {
        const wash = ctx.createRadialGradient(
          width * 0.52,
          laneY(1),
          50,
          width * 0.52,
          laneY(1),
          Math.max(width, height) * 0.7
        );
        wash.addColorStop(0, toRgba(palette.wash, 0.08));
        wash.addColorStop(1, "rgba(0,0,0,0)");
        ctx.fillStyle = wash;
        ctx.fillRect(0, 0, width, height);
      };

      const drawScene = (nowMs) => {
        ctx.clearRect(0, 0, width, height);
        drawBackgroundWash();
        drawLane();
        drawRelations(nowMs);
        drawStages();
        drawEffects();
        drawPackets();
      };

      const renderStaticScene = () => {
        packets = [];
        effects = [];
        rings = [];
        for (let i = 0; i < 18; i += 1) {
          const lane = i % laneRatios.length;
          const x = lerp(stageX(0) - 8, stageX(3) + 8, i / 17);
          packets.push({
            x,
            y: laneY(lane) + (lane - 1) * 2.5,
            lane,
            vx: 0,
            r: 1.9,
            alpha: 0.26,
            expanded: true,
            generated: true,
            ranked: x >= stageX(3)
          });
        }
        drawScene(0);
      };

      const frame = (time) => {
        if (!lastFrame) {
          lastFrame = time;
        }
        const dt = Math.min(0.034, (time - lastFrame) / 1000);
        lastFrame = time;

        spawnTimer += dt * 1000;
        if (spawnTimer >= 185) {
          spawnTimer = 0;
          spawnPacket();
        }

        updatePackets(dt, time);
        updateEffects(dt);
        drawScene(time);
        animationId = window.requestAnimationFrame(frame);
      };

      const resize = () => {
        width = window.innerWidth;
        height = window.innerHeight;
        dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
        canvas.width = width * dpr;
        canvas.height = height * dpr;
        canvas.style.width = `${width}px`;
        canvas.style.height = `${height}px`;
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      };

      const stop = () => {
        if (animationId) {
          window.cancelAnimationFrame(animationId);
          animationId = null;
        }
      };

      const start = () => {
        stop();
        updatePalette();
        resize();
        if (prefersReducedMotion.matches) {
          renderStaticScene();
          return;
        }
        packets = [];
        effects = [];
        rings = [];
        spawnTimer = 0;
        lastFrame = 0;
        for (let i = 0; i < 8; i += 1) {
          spawnPacket();
        }
        animationId = window.requestAnimationFrame(frame);
      };

      const onThemeMutation = () => {
        updatePalette();
        if (prefersReducedMotion.matches) {
          renderStaticScene();
        }
      };

      let resizeTimer;
      const onResize = () => {
        window.clearTimeout(resizeTimer);
        resizeTimer = window.setTimeout(() => {
          resize();
          if (prefersReducedMotion.matches) {
            renderStaticScene();
          }
        }, 100);
      };

      const onMotionPreferenceChange = () => {
        start();
      };

      window.addEventListener("resize", onResize, { passive: true });
      if (prefersReducedMotion.addEventListener) {
        prefersReducedMotion.addEventListener("change", onMotionPreferenceChange);
      } else if (prefersReducedMotion.addListener) {
        prefersReducedMotion.addListener(onMotionPreferenceChange);
      }

      const observer = new MutationObserver(onThemeMutation);
      observer.observe(root, { attributes: true, attributeFilter: ["data-theme"] });

      window.addEventListener(
        "pagehide",
        () => {
          stop();
          observer.disconnect();
        },
        { once: true }
      );

      start();
    })();
  </script>
</div>
