    <section id="specgoal-why" class="paper-section paper-anchor">
      <h2>Why it matters</h2>
      <ul class="paper-list">
        <li>DNN-based heuristics for planning typically assume a fixed, pre-defined goal.</li>
        <li>Changing the goal often requires retraining or enumerating all acceptable goal states.</li>
        <li>Many real problems require specifying properties of a goal rather than a single exact state.</li>
        <li>There is no formal mechanism to express logical goal constraints directly to a trained DNN.</li>
        <li>In domains with many valid target states (for example, Rubik's cube patterns), enumerating all goals is computationally burdensome.</li>
      </ul>
    </section>

    <section id="specgoal-what" class="paper-section paper-anchor">
      <h2>What we did</h2>
      <ul class="paper-list">
        <li>Introduced a goal-conditioned DQN, <code>Q(s, a, G)</code>, that estimates cost-to-go for a set of goal states.</li>
        <li>Represented goals as sets of ground atoms in first-order logic.</li>
        <li>Used Answer Set Programming (ASP) to generate stable models that define goal specifications.</li>
        <li>Trained with random walks (100-200 moves for Rubik's cube starts) and subsampled logical goal atoms.</li>
        <li>Combined the learned heuristic with weighted batched Q* search (batch size 10,000; weight 0.6).</li>
        <li>Demonstrated diverse Rubik's cube and Sokoban goals without retraining the DNN.</li>
        <li>Showed broader goal specifications can reduce solve time (for example, Cross6: 218.45 s vs canonical: 625.62 s).</li>
      </ul>
    </section>

    <section id="specgoal-how" class="paper-section paper-anchor">
      <h2>How it works</h2>
      <ul class="paper-list">
        <li>Goal-conditioned Q-learning: learn <code>Q(s, a, G)</code> where <code>G</code> is a set of ground atoms.</li>
        <li>Goal generation: convert a state to logical atoms and remove subsets to create generalized goals.</li>
        <li>ASP specification: use stable models from an ASP program to represent goal sets.</li>
        <li>Model refinement: if a stable model is not a valid goal model, iteratively expand it.</li>
        <li>Search: use batched weighted Q* search to reach states satisfying the logical goal.</li>
      </ul>

    <section id="specgoal-key" class="paper-section paper-anchor">
      <h2>Key contributions</h2>
      <ul class="paper-list">
        <li>Formalizes goal specification to DNN heuristics using first-order logic and ASP.</li>
        <li>Introduces a training procedure that generalizes across unseen goals without retraining.</li>
        <li>Demonstrates diverse Rubik's cube goals (for example, Cross6, CupSpot, Checkers) and Sokoban goals.</li>
        <li>Empirically shows that broader goal sets can reduce solve time and path cost (for example, Cross6 vs canonical).</li>
      </ul>

  <script>
    (() => {
      const attachCitationAnchor = () => {

      const canvas = document.getElementById("specgoal-logic-bg");
      if (!canvas) {
        return;
      }

      const ctx = canvas.getContext("2d", { alpha: true });
      if (!ctx) {
        return;
      }

      const root = document.documentElement;
      const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)");

      let width = 0;
      let height = 0;
      let dpr = 1;
      let palette = null;
      let geometry = null;
      let animationId = null;
      let lastFrame = 0;
      let sceneTime = 0;

      let packets = [];
      let cubePulses = [];
      const timers = { spawn: 0 };

      const parseColor = (value) => {
        if (!value) {
          return null;
        }
        const trimmed = value.trim();
        if (trimmed.startsWith("#")) {
          const hex = trimmed.slice(1);
          if (hex.length === 3) {
            return hex.split("").map((char) => parseInt(char + char, 16));
          }
          if (hex.length === 6) {
            return [
              parseInt(hex.slice(0, 2), 16),
              parseInt(hex.slice(2, 4), 16),
              parseInt(hex.slice(4, 6), 16)
            ];
          }
        }
        const rgbMatch = trimmed.match(/rgba?\(([^)]+)\)/);
        if (rgbMatch) {
          const parts = rgbMatch[1].split(",").map((part) => parseFloat(part.trim()));
          if (parts.length >= 3) {
            return parts.slice(0, 3);
          }
        }
        return null;
      };

      const toRgba = (rgb, alpha) => `rgba(${rgb[0]}, ${rgb[1]}, ${rgb[2]}, ${alpha})`;
      const mixColor = (base, accent, amount) =>

      const randInt = (min, max) => Math.floor(rand(min, max + 1));

      const clamp = (value, min, max) => Math.max(min, Math.min(max, value));
      const easeInOut = (value) => value * value * (3 - 2 * value);

      const updatePalette = () => {

        const accent = parseColor(styles.getPropertyValue("--global-link-color")) || [72, 146, 173];

          wash: mixColor(bg, accent, 0.2),
          lane: mixColor(bg, accent, 0.34),
          laneSoft: mixColor(bg, accent, 0.2),
          atom: mixColor(bg, accent, 0.68),
          accepted: mixColor(bg, accent, 0.8),
          rejected: mixColor(bg, text, 0.62),
          gate: mixColor(bg, accent, 0.58),
          gateSoft: mixColor(bg, accent, 0.36),
          cubeCell: mixColor(bg, accent, 0.4),
          cubeEdge: mixColor(bg, text, 0.52),
          cubePulse: mixColor(bg, accent, 0.9),
          label: mixColor(bg, text, 0.64)
        };
      };

      const buildGeometry = () => {
        const xLeft = width * 0.16;
        const xGate = width * 0.48;
        const xCube = width * 0.79;
        const yGate = height * 0.5;
        const laneOffsets = [-0.24, -0.12, 0, 0.12, 0.24];
        const lanes = laneOffsets.map((offset) => yGate + offset * height);

        geometry = {
          xLeft,
          xGate,
          xCube,
          yGate,
          lanes,
          cube: {
            x: xCube,
            y: yGate,
            cell: clamp(Math.min(width, height) * 0.036, 12, 24),
            gap: clamp(Math.min(width, height) * 0.006, 2, 4)
          }
        };
      };

      const resize = () => {
        width = window.innerWidth;
        height = window.innerHeight;
        dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
        canvas.width = width * dpr;
        canvas.height = height * dpr;
        canvas.style.width = `${width}px`;
        canvas.style.height = `${height}px`;
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
        buildGeometry();
      };

      const drawHexagon = (x, y, radius) => {
        for (let i = 0; i < 6; i += 1) {
          const angle = -Math.PI / 2 + (Math.PI / 3) * i;
          const px = x + Math.cos(angle) * radius;
          const py = y + Math.sin(angle) * radius;
          if (i === 0) {
            ctx.moveTo(px, py);
          } else {
            ctx.lineTo(px, py);
          }
        }
        ctx.closePath();
      };

      const spawnPacket = (seed = false) => {
        const lane = randInt(0, geometry.lanes.length - 1);
        const laneY = geometry.lanes[lane];
        packets.push({
          lane,
          laneY,
          state: "ingress",
          progress: seed ? rand(0, 1) : 0,
          speed: rand(0.5, 0.78),
          wobble: rand(0, Math.PI * 2),
          amp: rand(5, 12),
          radius: rand(1.5, 2.6),
          alpha: rand(0.24, 0.44),
          drift: rand(-30, 30),
          accepted: false
        });
      };

      const seedScene = () => {
        packets = [];
        cubePulses = [];
        for (let i = 0; i < 16; i += 1) {
          spawnPacket(true);
        }
        for (let i = 0; i < 5; i += 1) {
          cubePulses.push({ cell: randInt(0, 8), life: rand(0.22, 0.7), decay: rand(0.28, 0.42) });
        }
      };

      const emitCubePulse = () => {
        cubePulses.push({
          cell: randInt(0, 8),
          life: 1,
          decay: rand(0.78, 1.12)
        });
      };

      const packetPoint = (packet) => {
        const laneY = packet.laneY;
        const p = easeInOut(clamp(packet.progress, 0, 1));

        if (packet.state === "ingress") {
          const x = lerp(geometry.xLeft - 90, geometry.xGate, p);
          const sway = Math.sin(packet.wobble + packet.progress * 8) * packet.amp * (1 - p * 0.52);
          const y = lerp(laneY + packet.drift * 0.18, laneY, p) + sway;
          return { x, y, alpha: packet.alpha };
        }

        if (packet.state === "accepted") {
          const x = lerp(geometry.xGate, geometry.xCube - 76, p);
          const bridge = Math.sin(packet.wobble + packet.progress * 7) * (packet.amp * 0.5);
          const y = lerp(laneY, geometry.cube.y + packet.drift * 0.08, p) + bridge;
          return { x, y, alpha: packet.alpha + 0.08 };
        }

        const x = geometry.xGate + p * 88;
        const y = laneY + p * (68 + Math.abs(packet.drift) * 0.24);
        return { x, y, alpha: packet.alpha * (1 - p) };
      };

      const drawBackdrop = () => {
        const wash = ctx.createRadialGradient(
          geometry.xGate,
          geometry.yGate,
          20,
          geometry.xGate,
          geometry.yGate,

        );
        wash.addColorStop(0, toRgba(palette.wash, 0.11));
        wash.addColorStop(1, "rgba(0,0,0,0)");
        ctx.fillStyle = wash;
        ctx.fillRect(0, 0, width, height);
      };

      const drawLabels = () => {

          ctx.moveTo(geometry.xLeft - 86, laneY);
          ctx.quadraticCurveTo(

          ctx.moveTo(geometry.xLeft - 86, laneY);
          ctx.quadraticCurveTo(

          ctx.moveTo(geometry.xGate, geometry.yGate);
          ctx.quadraticCurveTo(
            geometry.xGate + 90,
            geometry.yGate + (laneIndex - 2) * 8,
            geometry.xCube - 84,
            geometry.cube.y + (laneIndex - 2) * 6
          );
          ctx.stroke();
        });
      };

      const drawGate = () => {

        ctx.strokeStyle = toRgba(palette.gateSoft, 0.26 + pulse * 0.14);

        ctx.beginPath();
        drawHexagon(geometry.xGate, geometry.yGate, inner);
        ctx.fill();

        ctx.strokeStyle = toRgba(palette.gate, 0.42);
        ctx.lineWidth = 1.2;
        ctx.beginPath();
        drawHexagon(geometry.xGate, geometry.yGate, inner + 5);
        ctx.stroke();
      };

      const drawRoundedRectPath = (x, y, w, h, r) => {
        const radius = Math.min(r, w * 0.5, h * 0.5);
        if (typeof ctx.roundRect === "function") {
          ctx.beginPath();
          ctx.roundRect(x, y, w, h, radius);
          return;
        }
        ctx.beginPath();
        ctx.moveTo(x + radius, y);
        ctx.lineTo(x + w - radius, y);
        ctx.quadraticCurveTo(x + w, y, x + w, y + radius);
        ctx.lineTo(x + w, y + h - radius);
        ctx.quadraticCurveTo(x + w, y + h, x + w - radius, y + h);
        ctx.lineTo(x + radius, y + h);
        ctx.quadraticCurveTo(x, y + h, x, y + h - radius);
        ctx.lineTo(x, y + radius);
        ctx.quadraticCurveTo(x, y, x + radius, y);
        ctx.closePath();
      };

      const drawCube = () => {
        const { x, y, cell, gap } = geometry.cube;
        const gridSize = cell * 3 + gap * 2;
        const startX = x - gridSize / 2;
        const startY = y - gridSize / 2;

        const panelPad = Math.max(8, cell * 0.38);
        ctx.fillStyle = toRgba(palette.cubeEdge, 0.08);
        ctx.strokeStyle = toRgba(palette.cubeEdge, 0.25);
        ctx.lineWidth = 1;
        drawRoundedRectPath(
          startX - panelPad,
          startY - panelPad,
          gridSize + panelPad * 2,

        );

        while (timers.spawn >= spawnEvery) {
          timers.spawn -= spawnEvery;
          if (packets.length < 90) {
            spawnPacket();
          }
        }

        packets = packets.filter((packet) => {
          packet.progress += packet.speed * (dt / 1000);

          if (packet.state === "ingress" && packet.progress >= 1) {
            packet.accepted = Math.random() < 0.68;
            packet.state = packet.accepted ? "accepted" : "rejected";
            packet.progress = 0;
            packet.speed = packet.accepted ? rand(0.46, 0.7) : rand(0.55, 0.85);
          } else if (packet.state === "accepted" && packet.progress >= 1) {
            emitCubePulse();
            return false;
          } else if (packet.state === "rejected" && packet.progress >= 1) {
            return false;
          }

          return true;
        });
      };

      const updateCubePulses = (dt) => {
        cubePulses = cubePulses.filter((pulse) => {
          pulse.life -= (dt / 1000) * pulse.decay;
          return pulse.life > 0.02;
        });
      };

      const drawPackets = () => {
        packets.forEach((packet) => {
          const point = packetPoint(packet);
          const color = packet.state === "rejected"
            ? palette.rejected
            : packet.state === "accepted"
              ? palette.accepted
              : palette.atom;
          ctx.fillStyle = toRgba(color, point.alpha);
          ctx.beginPath();
          ctx.arc(point.x, point.y, packet.radius, 0, Math.PI * 2);
          ctx.fill();
        });
      };

      const drawFrame = () => {
        ctx.clearRect(0, 0, width, height);
        drawBackdrop();
        drawLanes();
        drawGate();
        drawCube();
        drawPackets();
        drawLabels();
      };

      const start = () => {
        if (animationId) {
          window.cancelAnimationFrame(animationId);
          animationId = null;
        }

        timers.spawn = 0;
        sceneTime = 0;
        lastFrame = 0;
        updatePalette();
        resize();
        seedScene();
        drawFrame();

        if (!prefersReducedMotion.matches) {
          const loop = (timestamp) => {
            const dt = lastFrame ? Math.min(34, timestamp - lastFrame) : 16;
            lastFrame = timestamp;
            sceneTime += dt;
            updatePackets(dt);
            updateCubePulses(dt);
            drawFrame();
            animationId = window.requestAnimationFrame(loop);
          };
          animationId = window.requestAnimationFrame(loop);
        }
      };

      window.addEventListener(
        "resize",
        () => {
          resize();
          drawFrame();
        },
        { passive: true }
      );

      const handleMotionChange = () => {
        start();
      };

      if (prefersReducedMotion.addEventListener) {
        prefersReducedMotion.addEventListener("change", handleMotionChange);
      } else if (prefersReducedMotion.addListener) {
        prefersReducedMotion.addListener(handleMotionChange);
      }

      const observer = new MutationObserver(() => {
        updatePalette();
        drawFrame();
      });
      observer.observe(root, { attributes: true, attributeFilter: ["data-theme"] });

      start();
    })();
  </script>
</div>
