    <section id="pddl-why" class="paper-section paper-anchor">
      <h2>Why it matters</h2>
      <ul class="paper-list">
        <li>Planning models are typically trained and evaluated on a limited set of manually created or reconstructed domains.</li>
        <li>Limited domain diversity restricts generalization and encourages overfitting to benchmark distributions (for example, IPC-style domains).</li>
        <li>Domain reconstruction with LLMs often reproduces known domains rather than creating structurally novel ones.</li>
        <li>Reinforcement learning shows domain randomization improves robustness, but planning lacks an equivalent systematic mechanism.</li>
        <li>Robust evaluation of planners and planning foundation models requires scalable generation of diverse and structurally complex domains.</li>
      </ul>
    </section>

    <section id="pddl-what" class="paper-section paper-anchor">
      <h2>What we did</h2>
      <ul class="paper-list">
        <li>Introduced <strong>PDDLFuse</strong>, a tool that generates new PDDL domains by fusing two existing domains.</li>
        <li>Applied probabilistic modifications to action preconditions and effects (add, remove, and negate predicates) to control structural variation.</li>
        <li>Generated solvable problem instances by executing random action sequences from initial states.</li>
        <li>Evaluated solvability across increasing fusion depths (0-5), showing performance drops for LPG (10/10 to 4/10) and GOOSE (9/10 to 2/10), while the AVI-based model maintains 7/10 at Depth 5.</li>
        <li>Measured structural complexity using the k-WL test, where at Depth 5, 55% of domains require 3-WL and 20% require greater than 3-WL.</li>
      </ul>

    <section id="pddl-how" class="paper-section paper-anchor">
      <h2>How it works</h2>
      <ul class="paper-list">
        <li><strong>Domain fusion:</strong> merge objects, predicates, and actions from two base domains with systematic renaming to prevent overlap.</li>
        <li><strong>Probabilistic modification:</strong> add, remove, and negate predicates in preconditions and effects using configurable probabilities.</li>
        <li><strong>Problem generation:</strong> execute random action sequences to derive solvable goal states.</li>
        <li><strong>Complexity analysis:</strong> measure structural expressivity using k-WL tests across increasing depth levels.</li>
        <li><strong>Planner evaluation:</strong> test FD(FF), LPG, GOOSE, and an AVI-based model using Batch Weighted A* search.</li>
      </ul>
    </section>

    <section id="pddl-key" class="paper-section paper-anchor">
      <h2>Key contributions</h2>
      <ul class="paper-list">
        <li>A configurable tool for generating novel PDDL domains via domain fusion rather than reconstruction.</li>
        <li>Empirical evidence that increasing fusion depth reduces solvability for domain-independent planners (for example, LPG declines from 10/10 at Depth 1 to 4/10 at Depth 5).</li>
        <li>Demonstration that training on PDDLFuse-generated domains improves generalization for an AVI-based model compared to GOOSE at higher depths.</li>
        <li>WL-based expressivity analysis showing that at Depth 5, 55% of domains require 3-WL and 20% require greater than 3-WL, indicating increasing structural complexity.</li>
      </ul>

  <script>
    (() => {
      const attachCitationAnchor = () => {
        const citationParagraph = Array.from(document.querySelectorAll(".page__content p")).find(
          (paragraph) => paragraph.textContent.includes("Recommended citation")
        );
        if (citationParagraph && !citationParagraph.id) {
          citationParagraph.id = "paper-citation";
        }
      };

      attachCitationAnchor();
      window.requestAnimationFrame(attachCitationAnchor);
      if (document.readyState === "loading") {
        document.addEventListener("DOMContentLoaded", attachCitationAnchor, { once: true });
      }
      window.addEventListener("load", attachCitationAnchor, { once: true });

      const canvas = document.getElementById("pddlfuse-bg");
      if (!canvas) {
        return;
      }

      const ctx = canvas.getContext("2d");
      if (!ctx) {
        return;
      }

      const root = document.documentElement;
      const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)");

      let width = 0;
      let height = 0;
      let dpr = 1;
      let animationId = null;
      let palette = null;
      let nodes = { left: [], right: [], fused: [] };
      let particles = [];
      let pulse = 0;
      let lastTime = 0;

      const parseColor = (value) => {
        if (!value) {
          return null;
        }
        const trimmed = value.trim();
        if (trimmed.startsWith("#")) {
          const hex = trimmed.slice(1);
          if (hex.length === 3) {
            return hex.split("").map((char) => parseInt(char + char, 16));
          }
          if (hex.length === 6) {
            return [
              parseInt(hex.slice(0, 2), 16),
              parseInt(hex.slice(2, 4), 16),
              parseInt(hex.slice(4, 6), 16)
            ];
          }
        }
        const rgbMatch = trimmed.match(/rgba?\(([^)]+)\)/);
        if (rgbMatch) {
          const parts = rgbMatch[1].split(",").map((part) => parseFloat(part.trim()));
          if (parts.length >= 3) {
            return parts.slice(0, 3);
          }
        }
        return null;
      };

      const mixColor = (base, accent, amount) =>
        base.map((channel, index) => Math.round(channel + (accent[index] - channel) * amount));

      const toRgba = (rgb, alpha) => `rgba(${rgb[0]}, ${rgb[1]}, ${rgb[2]}, ${alpha})`;
      const rand = (min, max) => min + Math.random() * (max - min);

      const updatePalette = () => {
        const styles = getComputedStyle(root);
        const bgColor = parseColor(styles.getPropertyValue("--global-bg-color")) || [242, 242, 242];
        const accentColor = parseColor(styles.getPropertyValue("--global-link-color")) || [82, 173, 200];
        const textColor = parseColor(styles.getPropertyValue("--global-text-color")) || [55, 60, 68];

        palette = {
          wash: mixColor(bgColor, accentColor, 0.22),
          leftNode: mixColor(bgColor, accentColor, 0.65),
          rightNode: mixColor(bgColor, textColor, 0.56),
          fusedNode: mixColor(bgColor, accentColor, 0.76),
          linkLeft: mixColor(bgColor, accentColor, 0.45),
          linkRight: mixColor(bgColor, textColor, 0.42),
          linkMerge: mixColor(bgColor, accentColor, 0.58),
          particleA: mixColor(bgColor, accentColor, 0.8),
          particleB: mixColor(bgColor, textColor, 0.75),
          ring: mixColor(bgColor, accentColor, 0.64)
        };
      };

      const resetScene = () => {
        const leftX = width * 0.22;
        const rightX = width * 0.78;
        const centerX = width * 0.5;
        const centerY = height * 0.5;

        nodes.left = Array.from({ length: 11 }, (_, index) => ({
          x: leftX + rand(-60, 40),
          y: height * (0.16 + (index / 10) * 0.68) + rand(-16, 16),
          r: rand(2, 3.8)
        }));

        nodes.right = Array.from({ length: 11 }, (_, index) => ({
          x: rightX + rand(-40, 60),
          y: height * (0.16 + (index / 10) * 0.68) + rand(-16, 16),
          r: rand(2, 3.8)
        }));

        nodes.fused = Array.from({ length: 13 }, (_, index) => {
          const angle = (Math.PI * 2 * index) / 13 + rand(-0.14, 0.14);
          const radius = rand(34, 128);
          return {
            x: centerX + Math.cos(angle) * radius,
            y: centerY + Math.sin(angle) * (radius * 0.7),
            r: rand(1.8, 3.4),
            orbit: angle,
            speed: rand(-0.2, 0.2)
          };
        });

        particles = Array.from({ length: 30 }, (_, index) => {
          const fromLeft = index % 2 === 0;
          return {
            phase: rand(0, 1),
            speed: rand(0.13, 0.3),
            side: fromLeft ? "left" : "right",
            drift: rand(-22, 22),
            size: rand(1.1, 2.2)
          };
        });
      };

      const resize = () => {
        width = window.innerWidth;
        height = window.innerHeight;
        dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
        canvas.width = width * dpr;
        canvas.height = height * dpr;
        canvas.style.width = `${width}px`;
        canvas.style.height = `${height}px`;
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
        resetScene();
      };

      const drawLinks = (group, color, alpha) => {
        ctx.strokeStyle = toRgba(color, alpha);
        ctx.lineWidth = 1;
        for (let index = 0; index < group.length; index += 1) {
          const a = group[index];
          const b = group[index + 1];
          if (b) {
            ctx.beginPath();
            ctx.moveTo(a.x, a.y);
            ctx.lineTo(b.x, b.y);
            ctx.stroke();
          }
        }
      };

      const drawNodeSet = (group, color, alpha) => {
        ctx.fillStyle = toRgba(color, alpha);
        group.forEach((node) => {
          ctx.beginPath();
          ctx.arc(node.x, node.y, node.r, 0, Math.PI * 2);
          ctx.fill();
        });
      };

      const drawMergeRings = (now) => {
        const centerX = width * 0.5;
        const centerY = height * 0.5;
        for (let i = 0; i < 3; i += 1) {
          const radius = 48 + i * 28 + Math.sin(now * 0.0016 + i) * 6;
          ctx.strokeStyle = toRgba(palette.ring, 0.14 + i * 0.06);
          ctx.lineWidth = 1.2;
          ctx.beginPath();
          ctx.ellipse(centerX, centerY, radius, radius * 0.66, 0, 0, Math.PI * 2);
          ctx.stroke();
        }
      };

      const drawParticles = (dt) => {
        const centerX = width * 0.5;
        const centerY = height * 0.5;
        particles.forEach((particle, index) => {
          particle.phase += particle.speed * dt;
          if (particle.phase >= 1) {
            particle.phase = 0;
            particle.side = particle.side === "left" ? "right" : "left";
            particle.drift = rand(-22, 22);
          }

          const fromX = particle.side === "left" ? width * 0.22 : width * 0.78;
          const ctrlX = particle.side === "left" ? width * 0.38 : width * 0.62;
          const fromY = height * 0.15 + ((index % 15) / 14) * height * 0.7;
          const t = particle.phase;
          const inv = 1 - t;
          const x = inv * inv * fromX + 2 * inv * t * ctrlX + t * t * centerX;
          const y = inv * inv * fromY + 2 * inv * t * (centerY + particle.drift) + t * t * centerY;
          const color = particle.side === "left" ? palette.particleA : palette.particleB;

          ctx.fillStyle = toRgba(color, 0.23 + 0.4 * (1 - t));
          ctx.beginPath();
          ctx.arc(x, y, particle.size + (1 - t) * 0.8, 0, Math.PI * 2);
          ctx.fill();
        });
      };

      const render = (now) => {
        if (prefersReducedMotion.matches) {
          return;
        }

        if (!lastTime) {
          lastTime = now;
        }
        const dt = Math.min(0.05, (now - lastTime) / 1000);
        lastTime = now;

        ctx.clearRect(0, 0, width, height);

        const gradient = ctx.createRadialGradient(
          width * 0.5,
          height * 0.5,
          24,
          width * 0.5,
          height * 0.5,
          Math.max(width, height) * 0.7
        );
        gradient.addColorStop(0, toRgba(palette.wash, 0.12));
        gradient.addColorStop(1, "rgba(0, 0, 0, 0)");
        ctx.fillStyle = gradient;
        ctx.fillRect(0, 0, width, height);

        pulse += dt * 0.7;
        nodes.fused.forEach((node) => {
          node.orbit += node.speed * dt;
          const radiusX = 72 + 32 * Math.sin(pulse + node.orbit);
          const radiusY = 46 + 24 * Math.cos(pulse + node.orbit);
          node.x = width * 0.5 + Math.cos(node.orbit) * radiusX;
          node.y = height * 0.5 + Math.sin(node.orbit) * radiusY;
        });

        drawLinks(nodes.left, palette.linkLeft, 0.24);
        drawLinks(nodes.right, palette.linkRight, 0.24);

        ctx.strokeStyle = toRgba(palette.linkMerge, 0.2);
        ctx.lineWidth = 1;
        for (let i = 0; i < 6; i += 1) {
          const l = nodes.left[i * 2];
          const r = nodes.right[i * 2];
          const f = nodes.fused[(i * 2) % nodes.fused.length];
          if (!l || !r || !f) {
            continue;
          }
          ctx.beginPath();
          ctx.moveTo(l.x, l.y);
          ctx.lineTo(f.x, f.y);
          ctx.lineTo(r.x, r.y);
          ctx.stroke();
        }

        drawParticles(dt);
        drawMergeRings(now);
        drawNodeSet(nodes.left, palette.leftNode, 0.42);
        drawNodeSet(nodes.right, palette.rightNode, 0.4);
        drawNodeSet(nodes.fused, palette.fusedNode, 0.52);

        animationId = window.requestAnimationFrame(render);
      };

      const stop = () => {
        if (animationId) {
          window.cancelAnimationFrame(animationId);
          animationId = null;
        }
      };

      const start = () => {
        stop();
        if (prefersReducedMotion.matches) {
          ctx.clearRect(0, 0, width, height);
          return;
        }
        lastTime = 0;
        animationId = window.requestAnimationFrame(render);
      };

      const handleThemeOrResize = () => {
        updatePalette();
        resize();
        start();
      };

      handleThemeOrResize();

      window.addEventListener("resize", handleThemeOrResize, { passive: true });
      prefersReducedMotion.addEventListener("change", start);

      const observer = new MutationObserver(() => {
        updatePalette();
      });
      observer.observe(root, { attributes: true, attributeFilter: ["data-theme"] });
    })();
  </script>
</div>
